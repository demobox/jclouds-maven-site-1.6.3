49     private static final Set<Module> modules = ImmutableSet.<Module> of(
50           new ExecutorServiceModule(sameThreadExecutor(), sameThreadExecutor()));

52     static TerremarkECloudClient mockTerremarkECloudClient(String uri) {
53        Properties overrides = new Properties();
54        overrides.setProperty(PROPERTY_MAX_RETRIES, "1");
55        return ContextBuilder.newBuilder("trmk-ecloud")
56                             .credentials("user", "password")
57                             .endpoint(uri)
58                             .overrides(overrides)
59                             .modules(modules)
60                             .build(TerremarkECloudApiMetadata.CONTEXT_TOKEN).getApi();
61     }

63     String versionXML = "<SupportedVersions><VersionInfo><Version>0.8b-ext2.8</Version><LoginUrl>URLv0.8/login</LoginUrl></VersionInfo></SupportedVersions>";
64  
65     @Test
66     public void testLoginSetsContentLength() throws IOException, InterruptedException {
67        MockWebServer server = new MockWebServer();
68        AtomicReference<URL> url = setURLReplacingDispatcher(server);
69        server.enqueue(new MockResponse().setResponseCode(200).setBody(versionXML));
70        server.enqueue(new MockResponse().setResponseCode(200)
71                                         .addHeader("x-vcloud-authorization", "cookie")
72                                         .setBody("<OrgList />"));
73        server.play();
74        url.set(server.getUrl("/"));
75  
76        TerremarkECloudClient api = mockTerremarkECloudClient(url.get().toString());
77  
78        try {
79           api.listOrgs();
80        } finally {
81           RecordedRequest getVersions = server.takeRequest();
82           assertEquals(getVersions.getRequestLine(), "GET /versions HTTP/1.1");
83  
84           RecordedRequest login = server.takeRequest();
85           assertEquals(login.getRequestLine(), "POST /v0.8/login HTTP/1.1");
86           assertEquals(login.getHeader("Authorization"), "Basic dXNlcjpwYXNzd29yZA==");
87           assertEquals(login.getHeader("Content-Length"), "0");
88  
89           server.shutdown();
90        }
91     }

93     /**
94      * there's no built-in way to defer evaluation of a response header, hence this
95      * method, which allows us to send back links to the mock server.
96      */
97     private AtomicReference<URL> setURLReplacingDispatcher(MockWebServer server) {
98        final AtomicReference<URL> url = new AtomicReference<URL>();
99  
100       final QueueDispatcher dispatcher = new QueueDispatcher() {
101          protected final BlockingQueue<MockResponse> responseQueue = new LinkedBlockingQueue<MockResponse>();
102 
103          @Override
104          public MockResponse dispatch(RecordedRequest request) throws InterruptedException {
105             MockResponse response = responseQueue.take();
106             if (response.getBody() != null) {
107                String newBody = new String(response.getBody()).replace("URL", url.get().toString());
108                response = response.setBody(newBody);
109             }
110             return response;
111          }
112 
113          @Override
114          public void enqueueResponse(MockResponse response) {
115             responseQueue.add(response);
116          }
117       };
118       server.setDispatcher(dispatcher);
119       return url;
120    }
