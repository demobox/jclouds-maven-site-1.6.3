17  package org.jclouds.io.internal;
18  
19  import static com.google.common.base.Preconditions.checkArgument;
20  import static com.google.common.base.Preconditions.checkNotNull;
21  
22  import java.io.File;
23  import java.io.InputStream;
24  import java.io.IOException;
25  
26  import javax.inject.Singleton;
27  
28  import org.jclouds.io.Payload;
29  import org.jclouds.io.PayloadSlicer;
30  import org.jclouds.io.payloads.BaseMutableContentMetadata;
31  import org.jclouds.io.payloads.InputStreamPayload;
32  import org.jclouds.io.payloads.InputStreamSupplierPayload;
33  
34  import com.google.common.base.Throwables;
35  import com.google.common.io.ByteStreams;
36  import com.google.common.io.Files;
37  import com.google.common.io.InputSupplier;
38  
39  /**
40   * 
41   * @author Adrian Cole
42   */
43  @Singleton
44  public class BasePayloadSlicer implements PayloadSlicer {
45     /**
46      * {@inheritDoc}
47      */
48     @Override
49     public Payload slice(Payload input, long offset, long length) {
50        checkNotNull(input);
51        checkArgument(offset >= 0, "offset is negative");
52        checkArgument(length >= 0, "length is negative");
53        Payload returnVal;
54        if (input.getRawContent() instanceof File) {
55           returnVal = doSlice((File) input.getRawContent(), offset, length);
56        } else if (input.getRawContent() instanceof String) {
57           returnVal = doSlice((byte[]) input.getRawContent(), offset, length);
58        } else if (input.getRawContent() instanceof byte[]) {
59           returnVal = doSlice((byte[]) input.getRawContent(), offset, length);
60        } else if (input.getRawContent() instanceof InputStream) {
61           returnVal = doSlice((InputStream) input.getRawContent(), offset, length);
62        } else {
63           returnVal = doSlice(input, offset, length);
64        }
65        return copyMetadataAndSetLength(input, returnVal, length);
66     }
67  
68     protected Payload doSlice(Payload content, long offset, long length) {
69        return doSlice((InputSupplier<? extends InputStream>) content, offset, length);
70     }
71  
72     protected Payload doSlice(String content, long offset, long length) {
73        return doSlice(content.getBytes(), offset, length);
74     }
75  
76     protected Payload doSlice(File content, long offset, long length) {
77        return doSlice(Files.newInputStreamSupplier(content), offset, length);
78     }
79  
80     protected Payload doSlice(InputStream content, long offset, long length) {
81        try {
82           ByteStreams.skipFully(content, offset);
83        } catch (IOException ioe) {
84           throw Throwables.propagate(ioe);
85        }
86        return new InputStreamPayload(ByteStreams.limit(content, length));
87     }
88  
89     protected Payload doSlice(InputSupplier<? extends InputStream> content, long offset, long length) {
90        return new InputStreamSupplierPayload(ByteStreams.slice(content, offset, length));
91     }
92  
93     protected Payload doSlice(byte[] content, long offset, long length) {
94        Payload returnVal;
95        checkArgument(offset <= Integer.MAX_VALUE, "offset is too big for an array");
96        checkArgument(length <= Integer.MAX_VALUE, "length is too big for an array");
97        returnVal = new InputStreamSupplierPayload(
98              ByteStreams.newInputStreamSupplier(content, (int) offset, (int) length));
99        return returnVal;
100    }
101 
102    protected Payload copyMetadataAndSetLength(Payload input, Payload returnVal, long length) {
103       returnVal.setContentMetadata(BaseMutableContentMetadata.fromContentMetadata(input.getContentMetadata()
104             .toBuilder().contentLength(length).contentMD5(null).build()));
105       return returnVal;
106    }
107 
108 }
