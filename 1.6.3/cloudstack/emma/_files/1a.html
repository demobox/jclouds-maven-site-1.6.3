303   public void destroyNode(String id) {
304      String virtualMachineId = id;
305      // There was a bug in 2.2.12 release happening when static nat IP address
306      // was being released, and corresponding firewall rules were left behind.
307      // So next time the same IP is allocated, it might be not be static nat
308      // enabled, but there are still rules associated with it. And when you try
309      // to release this IP, the release will fail.
310      //
311      // The bug was fixed in 2.2.13 release only, and the current system wasn't
312      // updated yet.
313      //
314      // To avoid the issue, every time you release a static nat ip address, do
315      // the following:
316 
317      // 1) Delete IP forwarding rules associated with IP.
318      Set<String> ipAddresses = deleteIPForwardingRulesForVMAndReturnDistinctIPs(virtualMachineId);
319 
320      // 2) Delete firewall rules associated with IP.
321      ipAddresses.addAll(deleteFirewallRulesForVMAndReturnDistinctIPs(virtualMachineId));
322 
323      // 3) Disable static nat rule for the IP.
324      disableStaticNATOnIPAddresses(ipAddresses);
325 
326      // 4) Only after 1 and 2 release the IP address.
327      disassociateIPAddresses(ipAddresses);
328 
329      destroyVirtualMachine(virtualMachineId);
330 
331      vmToRules.invalidate(virtualMachineId);
332   }

373      for (IPForwardingRule rule : forwardingRules) {
374         if (!"Deleting".equals(rule.getState())) {
375            ipAddresses.add(rule.getIPAddressId());
376            String deleteForwardingRule = client.getNATClient().deleteIPForwardingRule(rule.getId());
377            if (deleteForwardingRule != null) {
378               logger.debug(">> deleting IPForwardingRule(%s) job(%s)", rule.getId(), deleteForwardingRule);
379               jobsToTrack.add(deleteForwardingRule);
380            }
381         }
382      }
383      awaitCompletion(jobsToTrack.build());
384      return ipAddresses;
385   }
