287   protected void cleanUpIncidentalResources(final String region, final String group){
288      // For issue #445, tries to delete security groups first: ec2 throws exception if in use, but
289      // deleting a key pair does not.
290      // This is "belt-and-braces" because deleteKeyPair also does extractIdsFromInstances & usingKeyPairAndNotDead
291      // for us to check if any instances are using the key-pair before we delete it. 
292      // There is (probably?) still a race if someone is creating instances at the same time as deleting them: 
293      // we may delete the key-pair just when the node-being-created was about to rely on the incidental 
294      // resources existing.
295 
296      // Also in #445, in aws-ec2 the deleteSecurityGroup sometimes fails after terminating the final VM using a 
297      // given security group, if called very soon after the VM's state reports terminated. Empirically, it seems that
298      // waiting a small time (e.g. enabling logging or debugging!) then the tests pass. We therefore retry.
299      // TODO: this could be moved to a config module, also the narrative above made more concise
300      retry(new Predicate<RegionAndName>() {
301         public boolean apply(RegionAndName input) {
302            try {
303               logger.debug(">> deleting incidentalResources(%s)", input);
304               deleteSecurityGroup(input.getRegion(), input.getName());
305               deleteKeyPair(input.getRegion(), input.getName()); // not executed if securityGroup was in use
306               logger.debug("<< deleted incidentalResources(%s)", input);
307               return true;
308            } catch (IllegalStateException e) {
309               logger.debug("<< inUse incidentalResources(%s)", input);
310               return false;
311            }
312         }
313      }, SECONDS.toMillis(3), 50, 1000, MILLISECONDS).apply(new RegionAndName(region, group));
314   }
